### 例1-3 生成排列数

分析：假如有abc三个字母，则它们的排列有：abc,acb,bac,bca,cba,cab。那么观察这个可以发现，分别固定三个字母，然后让剩余的字母排序就可以得到全部的排列顺序。

```cpp
#include <iostream>
#include <vector>
#include <iterator>     
#include <algorithm>
using namespace std;
template<class T>
void permutations(T* list, int begin, int end){
    if(begin == end){
        ostream_iterator<T> output(cout);
        copy(list,list+end,output);
        cout << endl;
    }else{
        for(int i = begin; i < end; ++i){
            swap(list[begin],list[i]);
            permutations(list,begin+1,end);
            swap(list[begin],list[i]);
        }
    }
}

template<class T>
void swap(T& val1, T& val2){
    T tmp = val1;
    val1 = val2;
    val2 = val1;
}

int main(){
    int list[6] = {1,2,3,4,5,6};
    permutations(list,0,6); 
    return 0;
}
```

### GCD 最大公约数

```cpp
int gcd(int x, int y){
    if(y == 0){
        return x;
    }else{
        return gcd(y,x % y);
    }
}


int main(){
    printf("%d",gcd(3,10));
    return 0;
}
```

### 子集生成

```cpp
#include <iostream>
using namespace std;

void print_sequence(int num, int len){
    if(!len){
        cout << endl;
        return;
    }
    int bit_num = (num & 0x00000001);
    num >>= 1;
    print_sequence(num,len-1);
    cout << bit_num;
}

int set_bit(int num, int index, bool open){
    if(open){
        num = num | (0x00000001 << index);
    }else{
        num = num & (0xffffffff ^ (0x00000001 << index));
    }
    return num;
}

void generate_subset(int nowlen,int len,int num = 0){
    if(nowlen == len - 1){
        print_sequence(set_bit(num,len-nowlen-1,true),len);
        print_sequence(set_bit(num,len-nowlen-1,false),len);
        return;
    }
    generate_subset(nowlen+1,len,set_bit(num,len-nowlen-1,true));
    generate_subset(nowlen+1,len,set_bit(num,len-nowlen-1,false));
}


int main(){
    generate_subset(0,4);
    return 0;
}
```
